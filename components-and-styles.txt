### src/components/NewsCard.js
```jsx
import React from 'react';

const NewsCard = ({ article }) => {
  // Format date for display
  const formatDate = (dateString) => {
    const date = new Date(dateString);
    return date.toLocaleDateString('en-US', {
      month: 'short',
      day: 'numeric',
      year: 'numeric',
      hour: '2-digit',
      minute: '2-digit'
    });
  };

  return (
    <article className="news-card">
      <div className="image-container">
        <img 
          src={article.imageUrl} 
          alt={article.title}
          onError={(e) => {
            e.target.onerror = null;
            e.target.src = 'https://via.placeholder.com/800x450?text=News';
          }}
        />
      </div>
      <div className="card-content">
        <div className="card-meta">
          <span className="positivity-score">
            {(article.positivityScore * 100).toFixed(0)}% Positive
          </span>
          <span className="publish-date">{formatDate(article.publishedAt)}</span>
        </div>
        <h3 className="card-title">{article.title}</h3>
        <p className="card-snippet">{article.snippet}</p>
        <div className="card-footer">
          <span className="source-name">{article.source}</span>
          <a 
            href={article.link}
            target="_blank"
            rel="noopener noreferrer"
            className="read-more-button"
          >
            Read More
          </a>
        </div>
      </div>
    </article>
  );
};

export default NewsCard;
```

### src/components/NewsGrid.js
```jsx
import React from 'react';
import NewsCard from './NewsCard';
import { FaNewspaper } from 'react-icons/fa';

const NewsGrid = ({ articles, loading, error, selectedSourcesCount }) => {
  if (loading) {
    return (
      <div className="loading-container">
        <div className="loader"></div>
        <p>Loading positive news for you...</p>
      </div>
    );
  }

  if (error) {
    return (
      <div className="error-container">
        <p className="error-message">{error}</p>
        <p>Please try again later or select different news sources.</p>
      </div>
    );
  }

  return (
    <div className="news-container">
      <div className="news-header">
        <h2>
          <FaNewspaper className="section-icon" />
          Your Positive News Feed
        </h2>
        <p className="news-stats">
          Showing {articles.length} positive stories from {selectedSourcesCount} sources
        </p>
      </div>

      {articles.length > 0 ? (
        <div className="news-grid">
          {articles.map(article => (
            <NewsCard key={article.id} article={article} />
          ))}
        </div>
      ) : (
        <div className="empty-state">
          <p>No articles match your current settings.</p>
          <p>Try selecting more news sources or lowering the positivity threshold.</p>
        </div>
      )}
    </div>
  );
};

export default NewsGrid;
```

### src/components/SettingsPanel.js
```jsx
import React from 'react';
import { FaCheck, FaTimes } from 'react-icons/fa';

const SettingsPanel = ({
  availableSources,
  selectedSources,
  toggleSource,
  positivityThreshold,
  setPositivityThreshold,
  setShowSettings
}) => {
  return (
    <div className="settings-panel">
      <div className="settings-header">
        <h2>News Feed Settings</h2>
        <button 
          onClick={() => setShowSettings(false)}
          className="close-settings"
          aria-label="Close settings panel"
        >
          <FaTimes />
        </button>
      </div>
      
      <div className="settings-section">
        <h3>News Sources</h3>
        <div className="sources-grid">
          {availableSources.map(source => (
            <div 
              key={source.id}
              className={`source-item ${selectedSources.includes(source.id) ? 'selected' : ''}`}
              onClick={() => toggleSource(source.id)}
            >
              <div className="checkbox">
                {selectedSources.includes(source.id) && <FaCheck />}
              </div>
              <span className="source-name">{source.name}</span>
              <span className="source-category">{source.category}</span>
            </div>
          ))}
        </div>
      </div>
      
      <div className="settings-section">
        <h3>Positivity Threshold</h3>
        <p className="slider-label">
          Only show articles with positivity score above: {(positivityThreshold * 100).toFixed(0)}%
        </p>
        <input
          type="range"
          min="0"
          max="1"
          step="0.05"
          value={positivityThreshold}
          onChange={(e) => setPositivityThreshold(parseFloat(e.target.value))}
          className="threshold-slider"
        />
        <div className="slider-labels">
          <span>Less Strict</span>
          <span>More Strict</span>
        </div>
      </div>
    </div>
  );
};

export default SettingsPanel;
```
